#!/bin/sh

docker run \\
  --volume=/var/run/docker.sock:/var/run/docker.sock \\
  --volume=/var/lib/drone:/data \\
  --env=DRONE_RPC_SERVER={{droneRpcServer}} \\
  --env=DRONE_RPC_SECRET={{droneRpcSecret}} \\
  --env=DRONE_RUNNER_CAPACITY={{droneRunnerCapacity}} \\
  --env=DRONE_LOGS_DEBUG={{droneLogsDebug}} \\
  --restart=always \\
  --detach=true \\
  --name=drone \\
  drone/agent:1.0.0-rc.1
